12 // Provides the way for plugin UI support implementation to get access to UI event loop
13 // so that it can appropriately dispatch some function invocation on the UI thread.
14 //
15 // Call `initializeOnUIThread()` once, and `start()` to run the main UI message loop.
16 // It will then go into infinite message loop until `stop()` is invoked by some event
17 // (or externally invoked on another thread).
18 //
19 // It is designed to be "pluggable" so that the actual hosting application can work with
20 // any UI framework that offers way to initialize event loop and dispatch UI events.
21 // Implement (derive from) this class and call `instance(EventLoop)` to use it.
22 // Make sure to do this before invoking `initializeOnUIThread()` (or any other functions).
23 //
24 // The default implementation (exists) is based on choc (`choc::gui::MessageLoop`).
25 class EventLoop {
26 protected:
27 virtual void initializeOnUIThreadImpl() = 0;
28 virtual bool runningOnMainThreadImpl() = 0;
29 virtual void enqueueTaskOnMainThreadImpl(std::function<void()>&& func) = 0;
30 virtual void startImpl() = 0;
31 virtual void stopImpl() = 0;
32 public:
33 virtual ~EventLoop() = default;
34
35 static void initializeOnUIThread() { getEventLoop()->initializeOnUIThreadImpl(); }
36 static bool runningOnMainThread() { return getEventLoop()->runningOnMainThreadImpl(); }
37 // Run task either immediately (if current thread is the main thread) or asynchronously (otherwise).
38 static void runTaskOnMainThread(std::function<void()>&& func) {
39 if (runningOnMainThread())
40 func();
41 else {
42 std::atomic<bool> done(false);
43 enqueueTaskOnMainThread([&done,func]() {
44 func();
45 done.store(true);
46 done.notify_one();
47 });
48 done.wait(false);
49 }
50 }
51 // Enqueue task on the main thread to run asynchronously.
52 static void enqueueTaskOnMainThread(std::function<void()>&& func) { getEventLoop()->enqueueTaskOnMainThreadImpl(std::move(func)); }
53 static void start() { getEventLoop()->startImpl(); }
54 static void stop() { getEventLoop()->stopImpl(); }
55 };
