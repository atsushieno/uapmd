15// ─── Handle ──────────────────────────────────────────────────────────────────
16
17// Opaque reference to a user-selected document.
18//
19// Desktop : id is the filesystem path string.
20// Android : id is the content:// URI string (permission already held).
21// Web : id is a path in Emscripten's virtual filesystem (MEMFS).
22//
23// Never construct a DocumentHandle manually; always obtain one from
24// IDocumentProvider::pickOpenDocuments, pickSaveDocument, or restoreHandle.
25struct DocumentHandle {
26 std::string id; // Platform-internal identifier — treat as opaque
27 std::string display_name; // Human-readable filename for UI labels
28 std::string mime_type; // MIME type if the platform reported it; else empty
29
30 bool valid() const { return !id.empty(); }
31};

33// ─── Filters ─────────────────────────────────────────────────────────────────
34
35// Describes a file-type category for the OS picker.
36// Populate both fields; each platform uses what it understands:
37// mime_types → Android and Web (accept attribute)
38// extensions → Desktop pfd (e.g. "*.wav")
39struct DocumentFilter {
40 std::string label;
41 std::vector<std::string> mime_types;
42 std::vector<std::string> extensions;
43};

62// ─── Interface ───────────────────────────────────────────────────────────────
63
64// Platform-agnostic interface for user-driven document access.
65//
66// Lifecycle: pick → read/write (and/or resolveToPath) → persistHandle (optional)
67//
68// Threading: Implementations may perform I/O on background threads internally,
69// but all callbacks are dispatched on the calling thread or the
70// platform's main/UI thread. Callers must not assume re-entrancy.
71//
72// Unsupported operations: all methods exist on every platform. Operations that
73// cannot be honoured invoke their callback with success == false and a
74// descriptive error string. They never throw.
75class IDocumentProvider {
76public:
77 virtual ~IDocumentProvider() = default;
78
79 static bool supportsCreateFileInternal();
80
81 using PickCallback = std::function<void(DocumentPickResult)>;
82 using ReadCallback = std::function<void(DocumentIOResult, std::vector<uint8_t>)>;
83 using WriteCallback = std::function<void(DocumentIOResult)>;
84 using PathCallback = std::function<void(DocumentIOResult, std::filesystem::path)>;
85
86 // ── Picking ──────────────────────────────────────────────────────────────
87
88 // Show the OS file-open picker.
89 // callback receives zero handles on cancel, one or more on success.
90 // allowMultiple requests multi-selection (best-effort; not guaranteed everywhere).
91 // On Android: URI permissions are taken inside the call before invoking callback.
92 virtual void pickOpenDocuments(
93 std::vector<DocumentFilter> filters,
94 bool allowMultiple,
95 PickCallback callback) = 0;
96
97 // Show the OS save-file picker / save dialog.
98 // callback receives exactly one handle on success, zero on cancel.
99 // defaultName: suggested filename (leaf name only; no path separators).
100 // On Web (no File System Access API): returns a handle immediately;
101 // writeDocument on that handle triggers a browser download instead.
102 virtual void pickSaveDocument(
103 std::string defaultName,
104 std::vector<DocumentFilter> filters,
105 PickCallback callback) = 0;
106
107 // ── I/O ──────────────────────────────────────────────────────────────────
108
109 // Read the complete document contents into memory.
110 // Suitable for project files and plugin state (typically ≤ a few MB).
111 virtual void readDocument(
112 DocumentHandle handle,
113 ReadCallback callback) = 0;
114
115 // Overwrite the document with new data.
116 virtual void writeDocument(
117 DocumentHandle handle,
118 std::vector<uint8_t> data,
119 WriteCallback callback) = 0;
120
121 // ── Path bridging ─────────────────────────────────────────────────────────
122
123 // Resolve a handle to a filesystem path usable with std::fstream and
124 // path-based library APIs (choc audio reader, UapmdProjectDataReader, etc.).
125 //
126 // Desktop : returns the path immediately — no copy performed.
127 // Android : streams the content URI to a temp file; returns that path.
128 // Web : the virtual FS path is already in handle.id; returns it directly.
129 //
130 // Temp files (when created) are valid until resolveToPath is called again
131 // for the same handle, or until the IDocumentProvider instance is destroyed.
132 // Callers must not cache the returned path beyond that scope.
133 //
134 // Prefer readDocument + in-memory parsing over this where possible.
135 virtual void resolveToPath(
136 DocumentHandle handle,
137 PathCallback callback) = 0;
138
139 // ── Main-thread pump ─────────────────────────────────────────────────────────
140
141 // Call once per frame from the application's main/render thread.
142 // Implementations that run dialogs asynchronously (e.g. desktop with a
143 // background pfd thread) use this to dispatch pending callbacks back to
144 // the main thread so that ImGui and audio-engine state can be touched safely.
145 // The default implementation is a no-op; override where needed.
146 virtual void tick() {}
147
148 // ── Persistence ───────────────────────────────────────────────────────────
149
150 // Serialize a handle into a storable string token (e.g. for a recent-files list).
151 //
152 // Desktop : token is the filesystem path.
153 // Android : token is the persisted content URI (permission already held).
154 // Web : returns "" — virtual FS paths do not survive a page reload.
155 virtual std::string persistHandle(const DocumentHandle& handle) = 0;
156
157 // Rebuild a handle from a token produced by persistHandle.
158 // Returns nullopt if the file no longer exists, access was revoked, or
159 // the token is from a different platform/session.
160 virtual std::optional<DocumentHandle> restoreHandle(const std::string& token) = 0;
161};
