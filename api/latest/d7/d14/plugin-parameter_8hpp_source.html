103 class ParameterEventBase {
104 public:
105 using EventListener = std::function<TReturn(TArgs...)>;
106
107 private:
108 std::atomic<EventListenerId> listenerIdCounter{1};
109 std::unordered_map<EventListenerId, EventListener> listeners{};
110 std::mutex listenerMutex{};
111
112 public:
113 EventListenerId addListener(EventListener listener) {
114 if (!listener)
115 return 0;
116 std::lock_guard<std::mutex> lock(listenerMutex);
117 auto id = listenerIdCounter++;
118 listeners.emplace(id, std::move(listener));
119 return id;
120 }
121
122 void removeListener(EventListenerId id) {
123 if (id == 0)
124 return;
125 std::lock_guard<std::mutex> lock(listenerMutex);
126 listeners.erase(id);
127 }
128
129 // FIXME: we should probably move it back to protected.
130 void notify(TArgs... args) {
131 std::vector<EventListener> callbacks;
132 {
133 std::lock_guard<std::mutex> lock(listenerMutex);
134 callbacks.reserve(listeners.size());
135 for (auto& kv : listeners)
136 callbacks.emplace_back(kv.second);
137 }
138 for (auto& cb : callbacks)
139 if (cb)
140 cb(args...);
141 }
142 };
