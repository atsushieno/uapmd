152 // Helper: Calculate absolute position from anchor
153 // For RT-safe queries, use TrackClipManager's cached resolution
154 TimelinePosition getAbsolutePosition(const std::unordered_map<int32_t, const ClipData*>& clips) const {
155 if (anchorClipId == -1) {
156 // Track anchor - anchorOffset is absolute position
157 // anchorOrigin is meaningless for track anchor (track start is always 0)
158 return anchorOffset;
159 }
160
161 // Clip anchor - resolve recursively
162 auto it = clips.find(anchorClipId);
163 if (it == clips.end()) {
164 // Fallback if anchor not found - treat as track anchor
165 return anchorOffset;
166 }
167
168 // Recursively resolve anchor chain
169 const ClipData* anchorClip = it->second;
170 TimelinePosition anchorClipPos = anchorClip->getAbsolutePosition(clips);
171
172 // Calculate anchor point based on origin
173 TimelinePosition anchorPoint;
174 if (anchorOrigin == AnchorOrigin::Start) {
175 anchorPoint = anchorClipPos; // Start of clip
176 } else {
177 // End of clip = start + duration
178 anchorPoint = anchorClipPos;
179 anchorPoint.samples += anchorClip->durationSamples;
180 }
181
182 return anchorPoint + anchorOffset;
183 }
